Item Trajectories

Explore how individual words are acquired over development. Pick a language and form, then select words to see the proportion of children reported to produce (or understand) each word at each age.

download_csv = (rows, filename) => {
  if (!rows || !rows.length) return html`<span></span>`;
  const cols = Object.keys(rows[0]);
  const esc = (v) => {
    const s = String(v ?? "");
    return /[",\n]/.test(s) ? '"' + s.replace(/"/g, '""') + '"' : s;
  };
  const csv = [cols.join(","), ...rows.map((r) => cols.map((c) => esc(r[c])).join(","))].join("\n");
  const url = URL.createObjectURL(new Blob([csv], { type: "text/csv" }));
  return html`<a class="btn btn-sm btn-outline-primary" href=${url} download=${filename}>Download CSV</a>`;
}
Plot = import("https://esm.sh/@observablehq/plot@0.6.17")
d3 = require("d3@7")

insts = transpose(instruments)
san = (s) => s.toLowerCase().replace(/[^a-z0-9]+/g, "_").replace(/^_+|_+$/g, "")
languages = [...new Set(insts.map((d) => d.language))].sort()
viewof language = Inputs.select(languages, {
  label: "Language",
  value: "English (American)"
})
viewof form = Inputs.select(
  insts.filter((d) => d.language === language).map((d) => d.form),
  { label: "Form" }
)
slice = d3.csv(`slices/items/${san(language + " " + form)}.csv`, d3.autoType)

form_type = insts.find((d) => d.language === language && d.form === form)?.form_type
measures = form_type === "WG" ? ["produces", "understands"] : ["produces"]
words = [...new Set(slice.map((d) => d.item_definition))].sort()
default_words = {
  const defaults = words.filter((w) => ["dog", "mommy*", "ball"].includes(w));
  return defaults.length ? defaults : words.slice(0, 3);
}
viewof measure = Inputs.select(measures, { label: "Measure" })
viewof selected = Inputs.select(words, {
  label: "Words",
  multiple: 8,
  value: default_words
})
trajectory_data = slice.filter(
  (d) => d.measure === measure && selected.includes(d.item_definition)
)
Plot.plot({
  style: { fontFamily: "var(--sans-serif)", fontSize: "13px" },
  width: 900,
  height: 450,
  inset: 8,
  grid: true,
  x: { label: "Age (months)", tickFormat: "" },
  y: { label: `Proportion of children who ${measure === "produces" ? "produce" : "understand"}`, domain: [0, 1] },
  color: { legend: true },
  marks: [
    Plot.dot(trajectory_data, {
      x: "age",
      y: "prop",
      stroke: "item_definition",
      r: (d) => Math.sqrt(d.n_children) / 4,
      fillOpacity: 0.5,
      tip: true
    }),
    Plot.lineY(
      trajectory_data,
      Plot.windowY(
        { k: 3, reduce: "mean" },
        { x: "age", y: "prop", stroke: "item_definition", curve: "basis", strokeWidth: 2.5 }
      )
    )
  ]
})
Inputs.table(
